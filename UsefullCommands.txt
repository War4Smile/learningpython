# Useful comands:

# Добавить текущий каталог в синхронизацию GIT (выполнятся на старте 1 раз)
git init

# Добавить в локальный ГИТ все файлы из директории (каждый раз при изменениях)
git add -A

# Добавить комментарий к выгрузке (каждый раз при изменениях)
git commit -m "Комментарий ..."



# Обновить репозиторий в GITHAB (по желанию синхронизировать с облаком)
git push -u origin main
